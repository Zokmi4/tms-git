Находим на docker hub образ nginx и скачиваем его 
docker run --name some-nginx -v /some/content:/usr/share/nginx/html:ro -d nginx
перезапускаем через приложение docker desktop и начинаем выполнять задание


Запуск контейнера на основе скаченного образа с docker hub
docker run -d \   # запуск контейнера в фоновом режиме 
  --name my_nginx_container \  
  -p 8080:80 \   # проброс портов 
  -e NGINX_HOST=localhost \ # локальное окружение 
  -e NGINX_PORT=80 \ 
  nginx

Создание Docker volume по команде docker volume create my_nginx_volume
Создаем контейнер с подключенным docker volume 
docker run -d \
  --name my_nginx_with_volume \
  -p 8081:80 \
  -v my_nginx_volume:/usr/share/nginx/html \
  nginx

Создаем сеть docker network create my_nginx_network

Создаем второй контейнер и подключаем к сети 
docker run -d \
  --name my_nginx_container2 \
  --network my_nginx_network \
  -v my_nginx_volume:/usr/share/nginx/html \
  nginx

Подключаемся к первому контейнеру nginx 
docker exec -it my_nginx_with_volume bash

Устанавливаем нано
apt-get update && apt-get install nano -y
Создаем файл и заполняем его рандомным текстом 
nano /usr/share/nginx/html/log.txt
При попутке создать запись в логах по команде logger -s "new message" >> /usr/share/nginx/html/log.txt выяснилось что  syslog у нас не установлен
т.к ошибка logger: socket /dev/log: Connection refused
По этому делаем запись в логи используя echo "new message" >> /usr/share/nginx/html/log.txt
